Emacs backup file

Write this in Emacs init file.

(defun -backup-file (file)
  "Append the date YYYY-MM-DD to a copy of FILE
FILE is the name of the file froms its absolute path."
  (interactive (list (when (buffer-file-name) (buffer-file-name))))
  (message file)
  (let ((backup (concat (file-name-directory file) (format-time-string "%Y-%m-%d") "--" (file-name-nondirectory file) "~")))
    (if (file-exists-p backup) (rename-file backup (concat "~/.local/share/Trash/files/" (file-name-nondirectory backup))) (copy-file file backup) (message (concat "Created: " backup)))))
